3. Installation Script (20 min)

File: scripts/install.sh

```bash
#!/bin/bash
# HFT System Installation Script

set -e

echo "=== HFT System Installation ==="

# Check prerequisites
check_prerequisites() {
    echo "Checking prerequisites..."

    # Check for required packages
    command -v cmake >/dev/null 2>&1 || { echo "cmake required"; exit 1; }
    command -v g++ >/dev/null 2>&1 || { echo "g++ required"; exit 1; }

    # Check kernel version
    kernel_version=$(uname -r)
    echo "Kernel version: $kernel_version"

    # Check CPU
    lscpu

    echo "Prerequisites OK"
}

# Install dependencies
install_dependencies() {
    echo "Installing dependencies..."
    sudo apt install -y libboost-all-dev libtbb-dev
    echo "Dependencies installed"
}

# Configure system
configure_system() {
    echo "Configuring system..."

    # CPU governor
    sudo ./scripts/setup_cpu_performance.sh

    # Hugepages
    sudo ./scripts/setup_hugepages.sh

    # IRQ affinity
    sudo ./scripts/setup_irq_affinity.sh

    echo "System configured"
}

# Install software
install_software() {
    echo "Installing HFT system..."

    # Create directories
    sudo mkdir -p /opt/hft
    sudo mkdir -p /var/log/hft
    sudo mkdir -p /etc/hft

    # Copy binaries
    sudo cp build/bin/* /opt/hft/

    # Copy configs
    sudo cp config/* /etc/hft/

    # Set permissions
    sudo chown -R $USER:$USER /opt/hft
    sudo chown -R $USER:$USER /var/log/hft

    echo "Installation complete"
}

# Main
check_prerequisites
install_dependencies
configure_system
install_software

echo "=== Installation Successful ==="
echo "Run: /opt/hft/hft_system --config /etc/hft/system_config.json"
```
