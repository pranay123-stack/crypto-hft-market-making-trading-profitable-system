================================================================================
KEY DESIGN PRINCIPLES
================================================================================

1. HEADER-ONLY VS COMPILED
   - Templates and inline functions: Header-only
   - Large implementations: Separate .cpp files
   - Public API: Headers in include/
   - Private implementation: Can stay in src/

2. NAMESPACE ORGANIZATION
   - Top-level: hft::
   - Sub-namespaces match directory structure:
     - hft::common::
     - hft::infrastructure::
     - hft::connectivity::
     - hft::market_data::
     - hft::trading::
     - hft::strategy::
     - hft::exchanges::
     - hft::persistence::
     - hft::utils::

3. FILE NAMING CONVENTIONS
   - Headers: snake_case.hpp
   - Implementation: snake_case.cpp
   - Test files: test_*.cpp or *_test.cpp
   - Benchmark files: bench_*.cpp

4. INCLUDE GUARDS
   - Use #pragma once (modern, simple)
   - Alternative: #ifndef HFT_PATH_FILE_HPP_

5. FORWARD DECLARATIONS
   - Use separate *_fwd.hpp headers for complex dependencies
   - Example: include/hft/trading/order_fwd.hpp

// include/hft/common/types.hpp
#pragma once

#include <cstdint>
#include <string>

namespace hft::common {

// Fixed-point price (8 decimal places)
class Price {
public:
    static constexpr int64_t SCALE = 100'000'000;

    constexpr Price() : value_(0) {}
    constexpr explicit Price(double d) : value_(static_cast<int64_t>(d * SCALE)) {}
    constexpr explicit Price(int64_t raw_value) : value_(raw_value) {}

    [[nodiscard]] constexpr double to_double() const {
        return static_cast<double>(value_) / SCALE;
    }

    [[nodiscard]] constexpr int64_t raw_value() const { return value_; }

    constexpr Price operator+(Price rhs) const { return Price(value_ + rhs.value_); }
    constexpr Price operator-(Price rhs) const { return Price(value_ - rhs.value_); }
    constexpr bool operator<(Price rhs) const { return value_ < rhs.value_; }
    constexpr bool operator>(Price rhs) const { return value_ > rhs.value_; }
    constexpr bool operator==(Price rhs) const { return value_ == rhs.value_; }

private:
    int64_t value_;
};

// Similarly for Quantity, Volume, etc.

using OrderId = std::string;
using Symbol = std::string;
using Exchange = std::string;

    BUY,

    STOP_LOSS_LIMIT,

    NEW,

    ACKNOWLEDGED,
    PARTIALLY_FILLED,
    FILLED,

    REJECTED,
    EXPIRED
};

#include <hft/infrastructure/logging/logger.hpp>
#include <hft/infrastructure/config/config_manager.hpp>
#include <hft/market_data/market_data_handler.hpp>
#include <hft/trading/oms/order_management_system.hpp>
#include <hft/strategy/strategy_engine.hpp>

int main(int argc, char** argv) {
    using namespace hft;

    // Initialize logger
    infrastructure::Logger::initialize("config/logging.yaml");

    // Load configuration
    infrastructure::ConfigManager config("config/production/config.yaml");

    // Create core components
    auto market_data = std::make_unique<market_data::MarketDataHandler>(config);
    auto oms = std::make_unique<trading::OrderManagementSystem>(config);
    auto strategy_engine = std::make_unique<strategy::StrategyEngine>(config);

    // Connect exchanges
    market_data->connect_exchanges();

    // Start strategy
    strategy_engine->start();

    // Main event loop
    while (true) {
        // Process events, handle signals, etc.
        std::this_thread::sleep_for(std::chrono::milliseconds(100));
    }

    return 0;
}
